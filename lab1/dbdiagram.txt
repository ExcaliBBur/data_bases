Table human {
  id integer [primary key]
  name text
  human_condition integer [ref: - human_condition.id]
  report integer [ref: <> report.id]
  spaceship integer [ref: <> spaceship.id]
}

Table human_condition {
  id integer [primary key]
  condition text
}

Table spaceship {
  id integer [primary key]
  name text
  type integer [ref: - spaceship_type.id]
  size text
}

Table flight {
  id integer [primary key]
  spaceship integer [ref: <> spaceship.id]
  space_object integer [ref: <> space_object.id]
}

Table space_object {
  id integer [primary key]
  figure text
  size text
}

Table spaceship_type {
  id integer [primary key]
  type text
}

Table report {
  id integer [primary key]
  frequency text
  content text
  destination text
}
